// Function to trim the string
function trim(s) {
  while (s.substring(0,1) == ' ') {
    s = s.substring(1,s.length);
  }
  while (s.substring(s.length-1,s.length) == ' ') {
    s = s.substring(0,s.length-1);
  }
  return s;
}

 function _HPEPKeycode_IE()
{
	var KeyValue=window.event.keyCode;
	//alert(KeyValue);
	//Considering only characters from a-z A-Z 1-9 0 @ - _ . , \ Enter space
	if((KeyValue >=97 && KeyValue <=122)||(KeyValue >=65 && KeyValue <=90)||(KeyValue >=49 && KeyValue <=57)||(KeyValue==48)||(KeyValue==42)||(KeyValue==64)||(KeyValue==46)||(KeyValue==92)||(KeyValue==45)||(KeyValue==95)||(KeyValue==13)||(KeyValue==44)||(KeyValue==32)||(KeyValue==40)||(KeyValue==41)||(KeyValue==43)||(KeyValue==47))
	{
		//Displays the entered Character
		return true;
	}
	else
	{
		//Do not Display the entered character
		event.keyCode=0;
		return false;
	}
	
}
